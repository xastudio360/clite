# ------------------------------------------------------------------------------
# Unity test framework
# Vendored under tests/vendor/unity/ or fetched via FetchContent.
# ------------------------------------------------------------------------------
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/vendor/unity/src/unity.c")
    # Vendored copy — preferred for reproducible builds
    add_library(unity STATIC
        vendor/unity/src/unity.c
    )
    target_include_directories(unity PUBLIC
        vendor/unity/src
    )
    message(STATUS "[CLite] Using vendored Unity test framework.")
else()
    # Fetch from GitHub if not vendored
    include(FetchContent)
    FetchContent_Declare(unity
        GIT_REPOSITORY https://github.com/ThrowTheSwitch/Unity.git
        GIT_TAG        v2.6.0
        GIT_SHALLOW    TRUE
    )
    FetchContent_MakeAvailable(unity)
    message(STATUS "[CLite] Fetched Unity test framework v2.6.0.")
endif()

# ------------------------------------------------------------------------------
# Helper function: clite_add_unit_test(<name> <source>)
#
# Creates a test executable, links it against CLite and Unity, and registers
# it with CTest. The test binary is named test_<name>.
# ------------------------------------------------------------------------------
function(clite_add_unit_test name source)
    set(_target "test_${name}")

    add_executable(${_target} "${source}")
    target_link_libraries(${_target} PRIVATE xastudio::clite unity)
    target_compile_options(${_target} PRIVATE
        $<$<C_COMPILER_ID:GNU,Clang,AppleClang>:-Wall -Wextra>
    )

    add_test(
        NAME    "unit::${name}"
        COMMAND ${_target}
    )
    set_tests_properties("unit::${name}" PROPERTIES
        LABELS          "unit"
        TIMEOUT         30
        FAIL_REGULAR_EXPRESSION "FAIL|ERROR"
        PASS_REGULAR_EXPRESSION "OK"
    )
endfunction()

# ------------------------------------------------------------------------------
# Helper function: clite_add_integration_test(<name> <source>)
# ------------------------------------------------------------------------------
function(clite_add_integration_test name source)
    set(_target "itest_${name}")

    add_executable(${_target} "${source}")
    target_link_libraries(${_target} PRIVATE xastudio::clite unity)

    add_test(
        NAME    "integration::${name}"
        COMMAND ${_target}
    )
    set_tests_properties("integration::${name}" PROPERTIES
        LABELS          "integration"
        TIMEOUT         60
        FAIL_REGULAR_EXPRESSION "FAIL|ERROR"
        PASS_REGULAR_EXPRESSION "OK"
    )
endfunction()
